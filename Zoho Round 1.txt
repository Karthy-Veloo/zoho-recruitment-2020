#include <iostream>
#include<string.h>

using namespace std;

int main()
{
    string in="WELCOME";
    int n=sizeof(in)/sizeof(in[0]);
    int p=n-1;
    string o;
    int j=0;
    for(int i=p/2;i<p;i++)
    {
        o[j]=in[i];
        for(int l=0;l<=j;l++)
        {
            cout<<o[l];
        }
        cout<<"\n";
        j++;
    }
    for(int i=0;i<p/2;i++)
    {
        o[j]=in[i];
        for(int l=0;l<=j;l++)
        {
            cout<<o[l];
        }
        cout<<"\n";
        j++;
    }

    return 0;
}
